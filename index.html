var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { var wykonawca = data[i][0] ? data[i][0].toString().trim() : ""; var album = data[i][1] ? data[i][1].toString().trim() : ""; var utwory = data[i][5]; // Kolumna F // Pobieraj tylko jeśli pole utwory jest puste if (!utwory && wykonawca && album) { try { // Czyścimy tytuł z dopisków w nawiasach dla lepszej skuteczności var czystyAlbum = album.replace(/\s*\(.*?\)\s*/g, '').replace(/\s*\[.*?\]\s*/g, '').trim(); var glownyWykonawca = wykonawca.split(',')[0].split('&')[0].trim(); // System kaskadowy (3 próby wyszukania) var queries = [ encodeURIComponent(wykonawca + " " + czystyAlbum), encodeURIComponent(glownyWykonawca + " " + czystyAlbum), encodeURIComponent(czystyAlbum) ]; var bestMatchId = null; for (var q = 0; q < queries.length; q++) { var urlAlbum = "https://itunes.apple.com/search?term=" + queries[q] + "&entity=album&limit=5&country=PL"; var resAlbum = UrlFetchApp.fetch(urlAlbum, {muteHttpExceptions: true}); if (resAlbum.getResponseCode() === 200) { var jsonAlbum = JSON.parse(resAlbum.getContentText()); if (jsonAlbum.results && jsonAlbum.results.length > 0) { if (q === 2) { // Jeśli szukamy po samym tytule, weryfikujemy artystę for (var j = 0; j < jsonAlbum.results.length; j++) { var apiArtist = jsonAlbum.results[j].artistName.toLowerCase(); if (apiArtist.includes(glownyWykonawca.toLowerCase().substring(0,4))) { bestMatchId = jsonAlbum.results[j].collectionId; break; } } } else { bestMatchId = jsonAlbum.results[0].collectionId; } } } if (bestMatchId) break; } if (bestMatchId) { var urlSongs = "https://itunes.apple.com/lookup?id=" + bestMatchId + "&entity=song&country=PL"; var resSongs = UrlFetchApp.fetch(urlSongs, {muteHttpExceptions: true}); if (resSongs.getResponseCode() === 200) { var jsonSongs = JSON.parse(resSongs.getContentText()); var tracklistArr = []; jsonSongs.results.forEach(function(item) { if (item.wrapperType === 'track') { var ms = item.trackTimeMillis || 0; var min = Math.floor(ms / 60000); var sec = ((ms % 60000) / 1000).toFixed(0); var timeStr = min + ":" + (sec < 10 ? '0' : '') + sec; tracklistArr.push(item.trackNumber + ". " + item.trackName + " (" + timeStr + ")"); } }); if (tracklistArr.length > 0) { sheet.getRange(i + 1, 6).setValue(tracklistArr.join('\n')); Logger.log("✅ POBRANO: " + album); } } } else { Logger.log("❌ NIE ZNALEZIONO: " + album); } Utilities.sleep(1000); // Oddech dla API Apple } catch (e) { Logger.log("⚠️ BŁĄD przy " + album + ": " + e.message); } } } }
